

#include <stdio.h>
#include <math.h>int main()
{
    int i = 0;
    for (i = 10000; i <= 99999; i++)
    {
        //判断i是否为lily number
        int sum = 0;
        int j = 0;
        //12345
        for (j = 1; j <= 4; j++)
        {
            int m = i % (int)pow(10, j);
            int n = i / (int)pow(10, j);
            sum += m * n;
        }
        if (sum == i)
        {
            printf("%d ", i);
        }
    }
    return 0;
}
